## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    filter: {
      invert: {
        property: "filter: invert(<v>)<i>;",

          none: 0,
        },
```
