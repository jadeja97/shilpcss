## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    live: {
      caret: {
        property: "caret-color: <v><i>;",
        resolve: "color",
        themeKey: "colors",
        variant: true,
        values: {
          auto: "auto",
        },
      },
    },
  },
};

export default shilpConfig;
```
