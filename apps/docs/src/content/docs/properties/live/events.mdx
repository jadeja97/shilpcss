## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    live: {
      events: {
        property: "pointer-events: <v><i>;",
        values: {
          // general
          auto: "auto",
          none: "none",

          // svg only
          all: "all",
          painted: "painted",
          fill: "fill",
          stroke: "stroke",
          box: "bounding-box",
          visible: {
            DEFAULT: "visible",
            painted: "visiblePainted",
            fill: "visibleFill",
            stroke: "visibleStroke",
          },
        },
      },
    },
  },
};

export default shilpConfig;
```
