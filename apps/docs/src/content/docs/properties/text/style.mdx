Set text style for text elements.

- `@text style-*;`

## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    text: {
      style: {
        property: "font-style: <v><i>;",
        values: {
          normal: "normal",
          italic: "italic",
        },
      },
    },
  },
};

export default shilpConfig;
```
