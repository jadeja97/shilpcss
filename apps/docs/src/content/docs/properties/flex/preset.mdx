## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    flex: {
      preset: {
        property: "flex: <v><i>;",
        values: {
          // flex: [grow] [shrink] [basis];
          1: "1 1 0%",
          auto: "1 1 auto",
          initial: "0 1 auto",
          none: "none",
        },
      },
    },
  },
};

export default shilpConfig;
```
