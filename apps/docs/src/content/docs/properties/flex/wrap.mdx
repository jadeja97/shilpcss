# Wrap Flex Items

Control flex wrapping behavior.

Usage:

- `@flex wrap;`
- `@flex wrap-reverse;`
- `@flex wrap-none;`
- `@flex wrap-*;`

<Alert variant="info">

This page contains only the **property configuration reference**. Complete documentation and usage
examples will be added progressively in future updates.

</Alert>

## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    flex: {
      wrap: {
        property: "flex-wrap: <v><i>;",
        values: {
          DEFAULT: "wrap",
          reverse: "wrap-reverse",
          none: "nowrap",
        },
      },
    },
  },
};

export default shilpConfig;
```
