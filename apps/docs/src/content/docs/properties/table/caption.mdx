# Table Caption Position

Set position of a table’s caption (top or bottom).

Usage:

- `@table caption-*;`

/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {

  properties: {

        property: "caption-side: <v><i>;",
