# Animation Mode

Set animation mode for fill behavior and end states.

Usage:

- `@animate mode-*;`

## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    animate: {
      mode: {
        property: "animation-fill-mode: <v><i>;",
        values: {
          backward: "backwards",
          forward: "forwards",
          both: "both",
          none: "none",
        },
      },
    },
  },
};

export default shilpConfig;
```
