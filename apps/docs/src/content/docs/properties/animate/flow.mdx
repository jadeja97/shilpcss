## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    animate: {
      flow: {
        property: "animation-timing-function: <v><i>;",
        themeKey: "flow",
        values: {
          linear: "linear",
        },
      },
    },
```
