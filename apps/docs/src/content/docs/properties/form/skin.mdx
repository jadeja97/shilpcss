## Reference

```js title="shilp.config.js"
/** @type {import('shilpcss/types').ShilpConfig} */
const shilpConfig = {
  source: "react",

  properties: {
    form: {
      skin: {
        property: "appearance: <v><i>;",
        values: {
          none: "none",
          auto: "auto",
          search: "search-field",
          textarea: "textarea",
          checkbox: "checkbox",
          radio: "radio",
          menu: "menulist",
          list: "listbox",
          meter: "meter",
          progress: "progress-bar",
          button: "button",
        },
      },
    },
  },
};

export default shilpConfig;
```
